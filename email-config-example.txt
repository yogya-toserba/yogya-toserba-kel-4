# Konfigurasi Email untuk Development

## Option 1: LOG (Development - Email tersimpan di log)
MAIL_MAILER=log
MAIL_HOST=127.0.0.1
MAIL_PORT=2525
MAIL_USERNAME=null
MAIL_PASSWORD=null
MAIL_ENCRYPTION=null
MAIL_FROM_ADDRESS="noreply@yogya.com"
MAIL_FROM_NAME="MyYOGYA"

## Option 2: Gmail SMTP (Production-like)

# MAIL_HOST=smtp.gmail.com

# MAIL_PASSWORD=your-app-password
